COLIMBOCAMOS
Sigue estos pasos una sola vez. Tarda menos de 5 minutos y es totalmente gratis.
1
Ve a sheets.new para crear una hoja de cálculo nueva de Google. Ponle el nombre que quieras (ej: Colimbocamos).
2
En la hoja, ve al menú Extensiones → Apps Script. Se abre un editor. Borra todo lo que hay y pega este código exactamente:
const SHEET_NAME = "Hoja 1"; function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); const headers = rows[0] || []; const data = rows.slice(1).map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const params = JSON.parse(e.postData.contents); if (params.action === "init") { if (sheet.getLastRow() === 0) { sheet.appendRow(["id","al","fe","pr","as","tp","sev","de","imgs","votos"]); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } if (params.action === "add") { const d = params.data; sheet.appendRow([d.id, d.al, d.fe, d.pr, d.as||"", d.tp, d.sev, d.de, JSON.stringify(d.imgs||[]), 0]); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } if (params.action === "vote") { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(params.id)) { const col = 10; // columna votos (índice 9 + 1) sheet.getRange(i+1, col).setValue(Number(rows[i][9]||0) + params.delta); break; } } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } if (params.action === "delete") { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(params.id)) { sheet.deleteRow(i+1); break; } } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ok:false})) .setMimeType(ContentService.MimeType.JSON); }
3
Pulsa el botón Implementar → Nueva implementación. En "Tipo" selecciona Aplicación web. En "Ejecutar como" pon Yo. En "Quién tiene acceso" pon Cualquier usuario. Pulsa Implementar y acepta los permisos.
4
Copia la URL de la aplicación web que aparece (empieza por https://script.google.com/macros/s/...) y pégala aquí abajo.
